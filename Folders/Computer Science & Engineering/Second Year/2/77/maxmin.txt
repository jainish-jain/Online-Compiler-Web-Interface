#include <iostream>
using namespace std;
int maxmin(int ,int ,int ,int )
{
int i=0,j=4,max,min,mid,max1,min1;
int a[5];
cout<<"enter elements"<<endl;
for(i=0;i<5;i++){
cin>>a[i];
}
    if(i==j)
    max=min=a[i];
    else if (i==j-1)
    {
	if(a[i]<a[j])
	{
	    max=a[j];
	}
	else
	{
	    max=a[i];
	    min=a[j];
	}
    }
    else
    {
	mid=((i+j)/2);
	maxmin(i,mid,max,min);
	maxmin(mid+1,j,max1,min1);
	if(max<max1)
	max=max1;
	if(min<min1)
	min=min1;

    }
    return 0;
}
